t = int(input())
for _ in range(t):
    n, k = map(int, input().split())
    arr = list(map(int, input().split()))
    count = 0
    i = 0
    while i <= n - k:
        if all(arr[j] == 0 for j in range(i, i + k)):
            count += 1
            i += k + 1  
        else:
            i += 1
    print(count)
